/************************************************************
 * Convert 24-bit ASPECT integer into 32-bit native integer *
 ************************************************************/
int conv_integer (void *pntr)
{
  unsigned      par;
  INTEGER       *par3000 = (INTEGER *)pntr;

  par = par3000->l | (par3000->m<<8) | (par3000->h<<16);
  if (par3000->h & 0x80) par |= 0xff000000;             /* negative ? */
  return (int)par;
}
\end{minted}
